11  import jakarta.enterprise.concurrent.ManagedScheduledExecutorService;
12  import jakarta.enterprise.concurrent.ManagedThreadFactory;
13  import jakarta.enterprise.context.ApplicationScoped;
14  import jakarta.enterprise.inject.Disposes;
15  import jakarta.enterprise.inject.Produces;
16  
17  @ApplicationScoped
18  class DefaultResources {
19  
20      @Produces
21      @TestScoped
22      ManagedScheduledExecutorService createDefaultService() {
23          return new ManagedScheduledExecutorServiceImpl("default", null, 0, false, 10, 0, SECONDS, 0, null,
24                  RejectPolicy.ABORT);
25      }
26  
27      void stopDefaultService(@Disposes ManagedScheduledExecutorService executorService) {
28          executorService.shutdownNow();
29      }
30  
31      @Produces
32      @TestScoped
33      ManagedThreadFactory createThreadFactory() {
34          return new ManagedThreadFactoryImpl("default");
35      }
36  
37      void stopDefaultService(@Disposes ManagedThreadFactory threadFactory) {
38          if (threadFactory instanceof ManagedThreadFactoryImpl) {
39              ((ManagedThreadFactoryImpl) threadFactory).stop();
40          }
41      }
42  }
