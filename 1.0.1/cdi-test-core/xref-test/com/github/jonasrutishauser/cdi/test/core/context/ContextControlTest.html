27  class ContextControlTest {
28  
29      @Inject
30      private RequestScopedBean requestScopedBean;
31      @Inject
32      private ApplicationScopedBean applicationScopedBean;
33      @Inject
34      private SessionScopedBean sessionScopedBean;
35      @Inject
36      private ConversationScopedBean conversationScopedBean;
37      @Inject
38      private ContextControl contextControl;
39  
40      @Inject
41      private RequestContextController requestContextController;
42  
43      @Test
44      void restartRequestStopAll() {
45          runTestStopAll(requestScopedBean);
46      }
47  
48      @Test
49      void restartSessionStopAll() {
50          runTestStopAll(sessionScopedBean);
51      }
52  
53      @Test
54      void restartConversationStopAll() {
55          requestContextController.deactivate();
56          requestContextController.activate();
57          UUID uuid = conversationScopedBean.getUuid();
58          requestContextController.deactivate();
59          requestContextController.activate();
60          UUID uuid2 = conversationScopedBean.getUuid();
61          assertEquals(uuid, uuid2);
62  
63          runTestStopAll(conversationScopedBean);
64      }
65  
66      @Test
67      void restartApplicationStopAll() {
68          TestInfo testInfo = contextControl.stop();
69          contextControl.start(testInfo);
70          UUID uuid = applicationScopedBean.getUuid();
71          contextControl.stop();
72          contextControl.start(testInfo);
73          UUID uuid2 = applicationScopedBean.getUuid();
74          assertEquals(uuid, uuid2);
75      }
76  
77      private void runTestStopAll(ScopedBean scopedBean) {
78          TestInfo testInfo = contextControl.stop();
79          contextControl.start(testInfo);
80          UUID uuid = scopedBean.getUuid();
81          contextControl.stop();
82          contextControl.start(testInfo);
83          UUID uuid2 = scopedBean.getUuid();
84          assertNotEquals(uuid, uuid2);
85      }
86  }
