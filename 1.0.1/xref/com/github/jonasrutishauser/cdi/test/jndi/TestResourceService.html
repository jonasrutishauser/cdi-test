18  public class TestResourceService extends AbstractResourceServices {
19  
20      static {
21          if (System.getProperty(INITIAL_CONTEXT_FACTORY) == null) {
22              System.setProperty(INITIAL_CONTEXT_FACTORY, org.osjava.sj.MemoryContextFactory.class.getName());
23              System.setProperty(SimpleJndi.SHARED, "true");
24              System.setProperty(SimpleJndi.JNDI_SYNTAX_SEPARATOR, "/");
25              System.setProperty(MemoryContext.IGNORE_CLOSE, "true");
26          }
27      }
28  
29      @Override
30      public Object resolveResource(InjectionPoint injectionPoint) {
31          try {
32              return super.resolveResource(injectionPoint);
33          } catch (IllegalArgumentException e) {
34              throw e;
35          } catch (RuntimeException e) {
36              if (e.getMessage().startsWith("Error looking up ") && injectionPoint.getType() instanceof Class) {
37                  Instance<?> instance = CDI.current().select((Class<?>) injectionPoint.getType());
38                  if (instance.isResolvable()) {
39                      return instance.get();
40                  }
41              }
42              throw e;
43          }
44      }
45  
46      @Override
47      protected String getResourceName(InjectionPoint injectionPoint) {
48          Resource resource = getResourceAnnotation(injectionPoint);
49          if (!resource.lookup().isBlank()) {
50              return resource.lookup();
51          }
52          return super.getResourceName(injectionPoint);
53      }
54  
55      @Override
56      protected Context getContext() {
57          try {
58              return new InitialContext();
59          } catch (NamingException e) {
60              throw new IllegalStateException();
61          }
62      }
63  
64  }
