18  /**
19   * Demo for running lots of (parameterized) junit 5 tests _fast_.
20   */
21  @ExtendWith(CdiTestJunitExtension.class)
22  class ParameterizedTestLifecycleTest {
23  
24      @Inject
25      private SampleService service;
26  
27      @Inject
28      private BackendServiceTestImplementation backendService;
29  
30      static Stream<String> createPersonNames() {
31          return Stream.
32          generate(UUID::randomUUID).
33          map(UUID::toString).
34          limit(5);
35      }
36  
37      @ParameterizedTest
38      @MethodSource("createPersonNames")
39      void storePerson(String name) {
40          Person testPerson = new Person(name);
41          service.storePerson(testPerson);
42          Assertions.assertEquals(1, backendService.getInvocations());
43      }
44  }
