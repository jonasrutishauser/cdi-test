19   * Test and demo lifecycle events.

21  @ExtendWith(CdiTestJunitExtension.class)
22  class NotificationLifecyleTest {
23  
24      @Inject
25      private NotificationLifecyleSupportBean supportBean;
26  
27      @Test
28      void notifyInitialized() {
29          assertNotNull(supportBean.initializedEvent);
30      }
31  
32      @Test
33      void notifyDestroyed() {
34          assertNotNull(supportBean.destroyedEvent);
35      }
36  
37      @Singleton
38      public static class NotificationLifecyleSupportBean {
39          TestInfo initializedEvent;
40          TestInfo destroyedEvent;
41  
42          protected void observeInitialized(@Observes @Initialized(TestScoped.class) TestInfo testEvent) {
43              this.initializedEvent = testEvent;
44          }
45  
46          protected void observeDestroyed(@Observes @Destroyed(TestScoped.class) TestInfo testEvent) {
47              this.destroyedEvent = testEvent;
48          }
49      }
