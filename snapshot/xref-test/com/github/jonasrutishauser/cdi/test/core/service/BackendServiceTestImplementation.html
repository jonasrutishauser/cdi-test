16  
17      private int invocations;
18  
19      private RuntimeException exceptionToThrow;
20  
21      @Override
22      public String storePerson(Person person) {
23          if (exceptionToThrow != null) {
24              throw exceptionToThrow;
25          } else {
26              invocations++;
27          }
28          return null;
29      }
30  
31      protected void testStarted(@Observes @Initialized(TestScoped.class) TestInfo testInfo) {
32          final BackendServiceException serviceException = testInfo.getTestMethod().getAnnotation(BackendServiceException.class);
33          if (serviceException != null) {
34              try {
35                  exceptionToThrow = serviceException.value()
36                          .getDeclaredConstructor()
37                          .newInstance();
38              } catch (InvocationTargetException | NoSuchMethodException | InstantiationException | IllegalAccessException e) {
39                  throw new RuntimeException("creating test exception failed", e);
40              }
41          }
42      }
43  
44      public int getInvocations() {
45          return invocations;
46      }
