14  /**
15   * Demo and test for the probably partly exotic case of an only partially
16   * complete test implementation:
17   * <p>
18   * Only the actually available methods will be redirected dynamically, all the
19   * others will throw an {@link IllegalStateException}. So for a test
20   * implementation you're not forced to implement all of the methods that you
21   * might not even expect to be used!
22   * </p>
23   */
24  @ExtendWith(CdiTestJunitExtension.class)
25  class ActivatePartialAlternativeForRegularBeanTest {
26  
27      @Inject
28      private SampleService sampleService;
29      @Inject
30      private BackendServiceTestPartialImplementation testBackendService;
31  
32      @Test
33      void callTestActivatedService() {
34          Person person = new Person();
35          Assertions.assertThrows(IllegalStateException.class, () -> sampleService.storePerson(person));
36      }
37  
38  }
