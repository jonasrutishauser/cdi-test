1   package com.github.jonasrutishauser.cdi.test.core.junit;

3   import java.lang.reflect.Method;
4   import java.util.Optional;
5   
6   import org.junit.jupiter.api.extension.AfterEachCallback;
7   import org.junit.jupiter.api.extension.AfterTestExecutionCallback;
8   import org.junit.jupiter.api.extension.BeforeEachCallback;
9   import org.junit.jupiter.api.extension.ExtensionContext;
10  import org.junit.jupiter.api.extension.ExtensionContext.Namespace;
11  import org.junit.jupiter.api.extension.ExtensionContext.Store;
12  import org.junit.jupiter.api.extension.InvocationInterceptor;
13  import org.junit.jupiter.api.extension.ReflectiveInvocationContext;
14  import org.junit.jupiter.api.extension.TestInstances;
15  
16  import com.github.jonasrutishauser.cdi.test.api.ImmutableTestInfo;
17  
18  public class CdiTestJunitExtension implements BeforeEachCallback, AfterTestExecutionCallback, AfterEachCallback, InvocationInterceptor {
19  
20      private static final Namespace NS = Namespace.create(CdiTestJunitExtension.class);
21  
22      @Override
23      public void beforeEach(ExtensionContext context) throws Exception {
24          CdiContainer cdiContainer = context.getRoot().getStore(NS).getOrComputeIfAbsent(CdiContainer.class);
25          cdiContainer.setTest(ImmutableTestInfo.builder().testClass(context.getRequiredTestClass())
26                  .testInstance(context.getRequiredTestInstance()).testMethod(context.getRequiredTestMethod())
27                  .testName(context.getDisplayName()).build());
28          inject(cdiContainer, context.getStore(NS), context.getRequiredTestInstances(), context.getRequiredTestClass(), TestMethodInterceptor.class);
29      }
30  
31      private void inject(CdiContainer cdiContainer, Store store, TestInstances testInstances, Class<?> testClass, Object storeKey) {
32          Optional<?> instance = testInstances.findInstance(testClass);
33          if (instance.isPresent()) {
34              store.put(storeKey, cdiContainer.inject(instance.get(), testClass));
35              if (testClass.getEnclosingClass() != null) {
36                  inject(cdiContainer, store, testInstances, testClass.getEnclosingClass(), new Object());
37              }
38          }
39      }
40  
41      @Override
42      public void interceptTestMethod(Invocation<Void> invocation, ReflectiveInvocationContext<Method> invocationContext,
43              ExtensionContext extensionContext) throws Throwable {
44          extensionContext.getStore(NS).get(TestMethodInterceptor.class, TestMethodInterceptor.class)
45                  .intercept(invocationContext.getExecutable(), invocationContext.getArguments(), invocation::proceed);
46      }
47      
48      @Override
49      public void afterTestExecution(ExtensionContext context) throws Exception {
50          context.getRoot().getStore(NS).get(CdiContainer.class, CdiContainer.class).testEnded();
51      }
52  
53      @Override
54      public void afterEach(ExtensionContext context) throws Exception {
55          context.getRoot().getStore(NS).get(CdiContainer.class, CdiContainer.class).clearTest();
56      }
57  
58  }
