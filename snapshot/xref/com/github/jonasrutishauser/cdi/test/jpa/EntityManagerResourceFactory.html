3   import java.lang.reflect.InvocationHandler;
4   import java.lang.reflect.InvocationTargetException;
5   import java.lang.reflect.Method;
6   import java.lang.reflect.Proxy;
7   
8   import org.jboss.logging.Logger;
9   import org.jboss.weld.injection.spi.ResourceReference;
10  import org.jboss.weld.injection.spi.ResourceReferenceFactory;
11  
12  import jakarta.enterprise.inject.spi.CDI;
13  import jakarta.persistence.EntityManager;
14  
15  public class EntityManagerResourceFactory implements ResourceReferenceFactory<EntityManager>, InvocationHandler {
16      private static final Logger LOG = Logger.getLogger(EntityManagerResourceFactory.class);
17  
18      private final String persistenceUnit;
19      private TestEntityResources testEntityResources;
20  
21  
22      public EntityManagerResourceFactory(String persistenceUnit) {
23          this.persistenceUnit = persistenceUnit;
24      }
25  
26      @Override
27      public ResourceReference<EntityManager> createResource() {
28          if (testEntityResources == null) {
29              testEntityResources = CDI.current().select(TestEntityResources.class).get();
30          }
31          return new ResourceReference<>() {
32              @Override
33              public EntityManager getInstance() {
34                  return (EntityManager) Proxy.newProxyInstance(EntityManager.class.getClassLoader(),
35                          new Class<?>[] {EntityManager.class}, EntityManagerResourceFactory.this);
36              }
37  
38              @Override
39              public void release() {
40                  LOG.trace("closing EntityManager will be done by TestEntityResources");
41              }
42          };
43      }
44      
45      @Override
46      public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
47          try {
48              return method.invoke(testEntityResources.resolveEntityManager(persistenceUnit), args);
49          } catch (InvocationTargetException e) {
50              throw e.getCause();
51          }
52      }
53  }
