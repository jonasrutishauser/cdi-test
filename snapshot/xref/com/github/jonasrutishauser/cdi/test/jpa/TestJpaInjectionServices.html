3   import java.util.Arrays;
4   import java.util.Map;
5   import java.util.stream.Collectors;
6   
7   import org.jboss.weld.injection.spi.JpaInjectionServices;
8   import org.jboss.weld.injection.spi.ResourceReferenceFactory;
9   
10  import jakarta.enterprise.inject.spi.InjectionPoint;
11  import jakarta.persistence.EntityManager;
12  import jakarta.persistence.EntityManagerFactory;
13  import jakarta.persistence.PersistenceContext;
14  import jakarta.persistence.PersistenceProperty;
15  import jakarta.persistence.PersistenceUnit;
16  
17  public class TestJpaInjectionServices implements JpaInjectionServices {
18      public static final String DEFAULT_TEST_PERSISTENCE_UNIT = "cdi-test";
19  
20      @Override
21      public ResourceReferenceFactory<EntityManager> registerPersistenceContextInjectionPoint(
22              InjectionPoint injectionPoint) {
23          PersistenceContext persistenceContext = injectionPoint.getAnnotated().getAnnotation(PersistenceContext.class);
24          if (persistenceContext == null) {
25              throw new IllegalStateException(
26                      "no @PersistenceContext annotation found on injection point " + injectionPoint);
27          }
28          String persistenceUnit = persistenceContext.unitName().isBlank() ? DEFAULT_TEST_PERSISTENCE_UNIT
29                  : persistenceContext.unitName();
30          TestEntityResources.addAdditionalProperties(persistenceUnit, convert(persistenceContext.properties()));
31          return new EntityManagerResourceFactory(persistenceUnit);
32      }
33  
34      private Map<String, String> convert(PersistenceProperty[] properties) {
35          return Arrays.stream(properties)
36                  .collect(Collectors.toMap(PersistenceProperty::name, PersistenceProperty::value));
37      }
38  
39      @Override
40      public ResourceReferenceFactory<EntityManagerFactory> registerPersistenceUnitInjectionPoint(
41              InjectionPoint injectionPoint) {
42          PersistenceUnit persistenceUnit = injectionPoint.getAnnotated().getAnnotation(PersistenceUnit.class);
43          if (persistenceUnit == null) {
44              throw new IllegalStateException(
45                      "no @PersistenceUnit annotation found on injection point " + injectionPoint);
46          }
47          return new EntityManagerFactoryResourceFactory(
48                  persistenceUnit.unitName().isBlank() ? DEFAULT_TEST_PERSISTENCE_UNIT : persistenceUnit.unitName());
49      }
50  
51      @Override
52      public void cleanup() {
53          // nothing
54      }
55  
56  }
