16  import jakarta.enterprise.context.spi.CreationalContext;
17  import jakarta.enterprise.inject.AmbiguousResolutionException;
18  import jakarta.enterprise.inject.Any;
19  import jakarta.enterprise.inject.spi.AnnotatedType;
20  import jakarta.enterprise.inject.spi.Bean;
21  import jakarta.enterprise.inject.spi.BeanManager;
22  import jakarta.enterprise.inject.spi.InjectionPoint;
23  import jakarta.enterprise.inject.spi.InjectionTarget;
24  import jakarta.inject.Singleton;
25  
26  class CdiContainer implements CloseableResource {
27  
28      private static final Logger LOGGER = Logger.getLogger(CdiContainer.class);
29  
30      private final Weld weld = new Weld();
31  
32      private WeldContainer weldContainer;
33  
34      public void setTest(TestInfo testInfo) {
35          if (weldContainer == null || !weldContainer.isRunning()) {
36              initialize(testInfo);
37          }
38          contextControl().start(testInfo);
39      }
40  
41      public void testEnded() {
42          contextControl().preStop();
43      }
44  
45      public void clearTest() {
46          contextControl().stop();
47      }
48  
49      public <T> TestMethodInterceptor inject(Object testInstance, Class<T> testClass) {
50          BeanManager beanManager = weldContainer.getBeanManager();
51          AnnotatedType<T> annotatedType = beanManager.createAnnotatedType(testClass);
52          Bean<T> bean;
53          try {
54              bean = (Bean<T>) beanManager.resolve(beanManager.getBeans(testClass, Any.Literal.INSTANCE));
55          } catch (AmbiguousResolutionException e) {
56              LOGGER.warnv("Failed to get test instance bean: {}", e.getMessage());
57              bean = null;
58          }
59          InjectionTarget<T> injectionTarget = beanManager.getInjectionTargetFactory(annotatedType)
60                  .createInjectionTarget(bean);
61          CreationalContext<T> ctx = beanManager.createCreationalContext(bean);
62          injectionTarget.inject(testClass.cast(testInstance), ctx);
63          for (InjectionPoint ip : injectionTarget.getInjectionPoints()) {
64              if (ip.getType() instanceof Class
65                      && ((Class<?>) ip.getType()).isAnnotationPresent(ActivatableTestImplementation.class)) {
66                  weldContainer.select(TestImplementationManager.class).get()
67                          .enableTestImplementation((Class<?>) ip.getType());
68              }
69          }
70          injectionTarget.postConstruct(testClass.cast(testInstance));
71          return new CdiTestMethodInterceptor<T>(weldContainer.getBeanManager(), testClass, testClass.cast(testInstance),
72                  injectionTarget, ctx);
73      }
74  
75      private ContextControl contextControl() {
76          return weldContainer.select(ContextControl.class).get();
77      }
78  
79      private void initialize(TestInfo testInfo) {
80          LOGGER.info("Booting CDI container");
81          long start = System.currentTimeMillis();
82          weld.addExtension(new CdiTestExtension(testInfo));
83          addBeans();
84          weldContainer = weld.initialize();
85          long end = System.currentTimeMillis();
86          LOGGER.infov("Booting CDI container finished in {0} ms", end - start);
87          if (!weldContainer.isRunning()) {
88              throw new IllegalStateException("Failed to start CDI container");
89          }
90      }
91  
92      private void addBeans() {
93          weld.addBeanDefiningAnnotations(Singleton.class, ExtendWith.class, Extensions.class);
94      }
95  
96      @Override
97      public void close() throws Throwable {
98          weld.shutdown();
99      }
100 }
