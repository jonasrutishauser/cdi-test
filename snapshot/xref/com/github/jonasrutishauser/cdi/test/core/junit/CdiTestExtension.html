62      private Set<String> readAllLines(String fileName) {
63          Set<String> result = new HashSet<>();
64          try {
65              Enumeration<URL> resources = getClass().getClassLoader().getResources(fileName);
66              while (resources.hasMoreElements()) {
67                  try (BufferedReader reader = new BufferedReader(
68                          new InputStreamReader(resources.nextElement().openStream()))) {
69                      for (String line = reader.readLine(); line != null; line = reader.readLine()) {
70                          if (!line.isBlank() && !line.trim().startsWith("#")) {
71                              result.add(line.trim());
72                          }
73                      }
74                  }
75              }
76          } catch (IOException e) {
77              throw new IllegalStateException(e);
78          }
79          return result;
80      }

82      void changeToTestScope(@Observes @WithAnnotations(ApplicationScoped.class) ProcessAnnotatedType<?> event) {
83          if (testScopedBeans.contains(event.getAnnotatedType().getJavaClass().getName())) {
84              AnnotatedTypeConfigurator<?> configurator = event.configureAnnotatedType()
85                      .remove(ApplicationScoped.class::isInstance).add(TestScoped.Literal.INSTANCE);
86              for (AnnotatedMethodConfigurator<?> method : configurator.methods()) {
87                  AnnotatedMethod<?> annotatedMethod = method.getAnnotated();
88                  if (!annotatedMethod.getParameters().isEmpty()
89                          && annotatedMethod.getParameters().get(0).isAnnotationPresent(Observes.class)) {
90                      if (annotatedMethod.getParameters().get(0).isAnnotationPresent(Initialized.class)) {
91                          method.params().get(0).remove(Initialized.class::isInstance)
92                                  .add(Initialized.Literal.of(TestScoped.class));
93                      } else if (annotatedMethod.getParameters().get(0).isAnnotationPresent(BeforeDestroyed.class)) {
94                          method.params().get(0).remove(BeforeDestroyed.class::isInstance)
95                                  .add(BeforeDestroyed.Literal.of(TestScoped.class));
96                      }
97                  }
98              }
99          }
100     }

109     void registerMockedTypes(@Observes @WithAnnotations({ExtendWith.class, Extensions.class}) ProcessAnnotatedType<?> event) {
110         try {
111             MockitoHelper helper = new MockitoHelper(mockedTypes);
112             event.getAnnotatedType().getFields().forEach(helper::registerMockedTypeIfNeeded);
113         } catch (NoClassDefFoundError e) {
114             // ignore as mockito is not in use
115         }
116     }

118     private static class MockitoHelper {

121         public MockitoHelper(Set<Type> mockedTypes) {
122             this.mockedTypes = mockedTypes;

125         private <T> void registerMockedTypeIfNeeded(AnnotatedField<T> field) {
126             if (field.isAnnotationPresent(Mock.class)) {
127                 mockedTypes.addAll(field.getTypeClosure());
128             }
129         }
130     }

132     void registerActivatableTestImplementation(
133             @Observes @WithAnnotations(ActivatableTestImplementation.class) ProcessAnnotatedType<?> event) {
134         ActivatableTestImplementation activatableTestImplementation = event.getAnnotatedType()
135                 .getAnnotation(ActivatableTestImplementation.class);
136         if (activatableTestImplementation != null) {
137             if (activatableTestImplementation.value().length == 0) {
138                 event.getAnnotatedType().getTypeClosure()
139                         .forEach(t -> registerTestImplementation(t, event.getAnnotatedType()));
140             } else {
141                 for (Class<?> type : activatableTestImplementation.value()) {
142                     registerTestImplementation(event.getAnnotatedType().getTypeClosure().stream()
143                             .filter(t -> type.equals(t) || (t instanceof ParameterizedType
144                                     && type.equals(((ParameterizedType) t).getRawType())))
145                             .findAny().orElse(type), event.getAnnotatedType());
146                 }
147             }
148         }
149     }

151     public void registerTestImplementation(Type type, AnnotatedType<?> testImplementation) {
152         if (!Object.class.equals(type)) {
153             testImplementations.computeIfAbsent(type, key -> new HashSet<>()).add(testImplementation.getJavaClass());
154         }
155     }

157     void configureTestBeans(@Observes ProcessBeanAttributes<?> event) {
158         if (!event.getAnnotated().getAnnotations(ExtendWith.class).isEmpty()
159                 && Dependent.class.equals(event.getBeanAttributes().getScope())
160                 && !event.getAnnotated().isAnnotationPresent(Dependent.class)) {
161             if (event.getAnnotated().getAnnotations(ExtendWith.class).stream().map(ExtendWith::value)
162                     .flatMap(Arrays::stream).anyMatch(isEqual(CdiTestJunitExtension.class))) {
163                 event.configureBeanAttributes().scope(TestScoped.class);
164             } else {
165                 event.veto();
166             }
167         }
168     }
169 
170     void configureActivatableTestImplementations(@Observes ProcessBeanAttributes<?> event) {
171         ActivatableTestImplementation activatableTestImplementation = event.getAnnotated()
172                 .getAnnotation(ActivatableTestImplementation.class);
173         if (activatableTestImplementation != null) {
174             event.configureBeanAttributes().types(event.getAnnotated().getBaseType());
175         } else if (event.getBeanAttributes().getTypes().stream().anyMatch(testImplementations.keySet()::contains)
176                 || event.getBeanAttributes().getTypes().stream()
177                         .anyMatch(isEqual(Object.class).negate().and(mockedTypes::contains))) {
178                             event.configureBeanAttributes().addStereotype(ReplaceableStereotype.class);
179                         }
180     }
181 
182     void enableGlobalTestImplementations(@Observes AfterTypeDiscovery event) {
183         event.getAlternatives().add(GlobalTestImplementation.class);
184     }

186     public Map<Class<?>, Set<Class<?>>> getTestImplementations(Bean<?> bean) {
187         return bean.getTypes().stream().filter(testImplementations::containsKey).collect(Collectors.toMap(
188                 t -> t instanceof ParameterizedType ? (Class<?>) ((ParameterizedType) t).getRawType() : (Class<?>) t,
189                 testImplementations::get));
190     }
