3   import java.util.HashMap;

5   import org.jboss.weld.context.bound.BoundConversationContext;
6   import org.jboss.weld.context.bound.BoundSessionContext;
7   import org.jboss.weld.context.bound.MutableBoundRequest;
8   
9   import com.github.jonasrutishauser.cdi.test.api.TestInfo;
10  import com.github.jonasrutishauser.cdi.test.api.context.TestScoped;
11  
12  import jakarta.enterprise.context.ApplicationScoped;
13  import jakarta.enterprise.context.BeforeDestroyed;
14  import jakarta.enterprise.context.Destroyed;
15  import jakarta.enterprise.context.Initialized;
16  import jakarta.enterprise.context.control.RequestContextController;
17  import jakarta.enterprise.event.Event;
18  import jakarta.enterprise.inject.spi.BeanManager;
19  import jakarta.inject.Inject;
20  
21  @ApplicationScoped
22  public class ContextControl {
23  
24      private final TestContext testContext;
25      private final Event<TestInfo> event;
26  
27      private final BoundSessionContext sessionContext;
28      private final BoundConversationContext conversationContext;
29      private final RequestContextController requestContextController;
30  
31      ContextControl() {
32          this(null, null, null, null, null);
33      }
34      
35      @Inject
36      ContextControl(BeanManager beanManager, Event<TestInfo> event, BoundSessionContext sessionContext,
37              BoundConversationContext conversationContext, RequestContextController requestContextController) {
38          this.testContext = (TestContext) beanManager.getContext(TestScoped.class);
39          this.event = event;
40          this.sessionContext = sessionContext;
41          this.conversationContext = conversationContext;
42          this.requestContextController = requestContextController;
43      }
44  
45      public void start(TestInfo testInfo) {
46          testContext.setTestInfo(testInfo);
47          event.select(Initialized.Literal.of(TestScoped.class)).fire(testInfo);
48          HashMap<String, Object> sessionStore = new HashMap<>();
49          sessionContext.associate(sessionStore);
50          sessionContext.activate();
51          conversationContext.associate(new MutableBoundRequest(new HashMap<>(), sessionStore));
52          conversationContext.activate();
53          requestContextController.activate();
54      }
55  
56      public void preStop() {
57          event.select(BeforeDestroyed.Literal.of(TestScoped.class)).fire(testContext.getTestInfo());
58      }
59  
60      public TestInfo stop() {
61          requestContextController.deactivate();
62          conversationContext.deactivate();
63          sessionContext.invalidate();
64          sessionContext.deactivate();
65          sessionContext.dissociate(null);
66          TestInfo testInfo = testContext.clearTestInfo();
67          event.select(Destroyed.Literal.of(TestScoped.class)).fire(testInfo);
68          return testInfo;
69      }
70  
71  }
